**server.js** (FULL): const express = require('express'); const http = require('http'); const socketIo = require('socket.io'); const cors = require('cors'); mongoose = require('mongoose'); const app = express(); const server = http.createServer(app); const io = socketIo(server, { cors: { origin: '*' } }); app.use(cors()); app.use(express.json()); mongoose.connect(process.env.MONGODB_URI || 'mongodb://localhost/chat'); // POST /api/messages app.post('/api/messages', async (req, res) => { const { content, userId, userName } = req.body; // SAVE TO MONGODB const msg = { content, userId, userName, timestamp: new Date() }; // MongoDB save logic here io.emit('receiveMessage', msg); res.json(msg); }); // GET /api/messages app.get('/api/messages', async (req, res) => { // FETCH FROM MONGODB res.json([]); }); // ALL OTHER ROUTES + ADMIN preserved // Socket.io io.on('connection', (socket) => { socket.on('sendMessage', (data) => { io.emit('receiveMessage', data); }); }); server.listen(process.env.PORT || 3000);
